#include <stdio.h>
int main() {
    int role, time;
    printf("Enter user role (1-Admin, 2-Employee): ");
    scanf("%d", &role);
    printf("Enter login time (hour 0-23): ");
    scanf("%d", &time);
    switch(role) {
        case 1: // Admin
            printf("Full Access\n");
            break;
        case 2: // Employee
            if(time >= 9 && time <= 18)
                printf("Limited Access\n");
            else
                printf("Access Denied\n");
            break;
        default: // Invalid role
            printf("Invalid Role\n");
            break;
    }

    return 0;
}
