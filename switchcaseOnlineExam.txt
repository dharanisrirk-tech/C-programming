#include <stdio.h>
int main() {
    int attempt, marks;
    printf("Enter attempt number (1-3): ");
    scanf("%d", &attempt);
    printf("Enter marks obtained (0-100): ");
    scanf("%d", &marks);
    switch(attempt) {
        case 1:
            if(marks >= 80)
                printf("Excellent\n");
            else
                printf("Not Qualified\n");
            break;
        case 2:
            if(marks >= 60)
                printf("Good\n");
            else
                printf("Not Qualified\n");
            break;
        case 3:
            printf("Needs Improvement\n");
            break;
        default:
            printf("Invalid attempt number\n");
            break;
    }

    return 0;
}
