#include <stdio.h>
int main() {
    int vehicleType, hours;
    int rate, totalFee;
      printf("Enter vehicle type (1 for Bike, 2 for Car): ");
    scanf("%d", &vehicleType);
    printf("Enter number of hours parked: ");
    scanf("%d", &hours);
    switch(vehicleType) {
        case 1: // Bike
            rate = 10;
            break;
        case 2: // Car
            rate = 20;
            break;
        default: // Invalid vehicle type
            rate = 20;
            break;
    }
    totalFee = rate * hours;
    printf("Parking Fee â‚¹%d\n", totalFee);

    return 0;
}
